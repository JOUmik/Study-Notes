## 427. 建立四叉树

**关键词**：分治法

~~~c++
/*
// Definition for a QuadTree node.
class Node {
public:
    bool val;
    bool isLeaf;
    Node* topLeft;
    Node* topRight;
    Node* bottomLeft;
    Node* bottomRight;
    
    Node() {
        val = false;
        isLeaf = false;
        topLeft = NULL;
        topRight = NULL;
        bottomLeft = NULL;
        bottomRight = NULL;
    }
    
    Node(bool _val, bool _isLeaf) {
        val = _val;
        isLeaf = _isLeaf;
        topLeft = NULL;
        topRight = NULL;
        bottomLeft = NULL;
        bottomRight = NULL;
    }
    
    Node(bool _val, bool _isLeaf, Node* _topLeft, Node* _topRight, Node* _bottomLeft, Node* _bottomRight) {
        val = _val;
        isLeaf = _isLeaf;
        topLeft = _topLeft;
        topRight = _topRight;
        bottomLeft = _bottomLeft;
        bottomRight = _bottomRight;
    }
};
*/

class Solution {
public:
    Node* construct(vector<vector<int>>& grid) {
        int x = grid.size();
        int y = grid[0].size();
        Node* res = divideAndConquer(grid, 0, x - 1, 0, y - 1);

        return res;
    }

    Node* divideAndConquer(vector<vector<int>>& grid, int x_begin, int x_end, int y_begin, int y_end){
        int temp = grid[x_begin][y_begin];
        int flag = 1;   //1代表不需要继续拆分，0代表需要拆分

        Node* res;



        for(int i = x_begin; i <= x_end; i++){
            if(flag == 0) break;
            for(int j = y_begin; j <= y_end; j++){
                if(temp != grid[i][j]){
                    flag = 0;
                    break;
                }
            }
        }

        if(flag == 1){
            res = new Node(temp, true);
        }
        else{
            Node* topLeft = divideAndConquer(grid, x_begin, (x_end - x_begin + 1) / 2 + x_begin - 1, y_begin, (y_end - y_begin + 1) / 2 + y_begin - 1);
            Node* topRight = divideAndConquer(grid, x_begin, (x_end - x_begin + 1) / 2 + x_begin - 1, (y_end - y_begin + 1) / 2 + y_begin, y_end);
            Node* bottomLeft = divideAndConquer(grid, (x_end - x_begin + 1) / 2 + x_begin, x_end, y_begin, (y_end - y_begin + 1) / 2 + y_begin - 1);
            Node* bottomRight = divideAndConquer(grid, (x_end - x_begin + 1) / 2 + x_begin, x_end, (y_end - y_begin + 1) / 2 + y_begin, y_end);

            res = new Node(temp, false, topLeft, topRight, bottomLeft, bottomRight);
        }

        return res;
    }
};
~~~



## 23. 合并K个升序链表

**关键词**：虚头节点

~~~c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if(lists.empty()){
            return nullptr;
        }

        int valid_i = 0;

        for(int i = 0; i < lists.size(); i++){
            if(i == valid_i) continue;
            lists[valid_i] = merge(lists[valid_i], lists[i]);
        }

        return lists[valid_i];
    }

    ListNode* merge(ListNode* A, ListNode* B){
        if(A == nullptr){
            A = B;
            return B;
        }
        
        if(B == nullptr){
            return A;
        }

        ListNode* nodeA = new ListNode(INT_MIN, A);
        ListNode* root = nodeA;
        ListNode* nodeB = B;

        while(nodeB != nullptr){
            if(nodeA == nullptr){
                nodeA = nodeB;
                break;
            }
            else if(nodeA->val <= nodeB->val && nodeA->next == nullptr){
                nodeA->next = nodeB;
                break;
            }
            else if(nodeA->val <= nodeB->val && nodeA->next->val > nodeB->val){
                ListNode* temp = nodeA->next;
                nodeA->next = nodeB;
                nodeB = nodeB->next;
                nodeA->next->next = temp;
                nodeA = nodeA->next;
            }
            else{
                nodeA = nodeA->next;
            }
        }

        return root->next;
    }
};
~~~
