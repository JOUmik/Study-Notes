## 211. 添加和搜索单词

**关键词**：字典树， DFS

~~~C++
class WordDictionary {
public:
    WordDictionary() {
        children.resize(26);
        isEnd = false;
    }
    
    void addWord(string word) {
        WordDictionary* node = this;
        for(char ch : word){
            if(node->children[ch - 'a'] == nullptr){
                node->children[ch - 'a'] = new WordDictionary();
            }
            node = node->children[ch - 'a'];
        }

        node->isEnd = true;
    }
    
    bool search(string word) {
        return dfs(word, 0, this);
    }

    bool dfs(const string & word,int index,WordDictionary * node) {
　　　　if (index == word.size()) {
            return node->isEnd;    
        }

        char ch = word[index];
        if (ch >= 'a' && ch <= 'z') {
            WordDictionary * child = node->children[ch - 'a'];
            if (child != nullptr && dfs(word, index + 1, child)) {
                return true;
            }
        } else if (ch == '.') {
            for (int i = 0; i < 26; i++) {
                WordDictionary * child = node->children[i];
                if (child != nullptr && dfs(word, index + 1, child)) {
                    return true;
                }
            }
        }
        return false;
    }

private:
    vector<WordDictionary*> children;
    bool isEnd;
};

/**
 * Your WordDictionary object will be instantiated and called as such:
 * WordDictionary* obj = new WordDictionary();
 * obj->addWord(word);
 * bool param_2 = obj->search(word);
 */
~~~

## 212. 单词搜索II

**关键词**：字典树， DFS

~~~c++
class Solution {
public:
    int dirs[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    Trie* t = new Trie();
    vector<string> res;
    int l = 0;

    vector<string> findWords(vector<vector<char>>& board, vector<string>& words) {
        //先把所有字符串放入字典树
        for(int i = 0; i<words.size(); i++){
            t->insert(words[i]);
            if(words[i].length()>l) l = words[i].length();
        }
        
        for(int i = 0; i < board.size(); i++){
            for(int j = 0; j < board[0].size(); j++){
                string s = "";
                dfs(board, i, j, s, t, 0);
            }
        }

        return res;
    }

    //能用&就用，能极大减少时间开销
    void dfs(vector<vector<char>>& board, int x, int y, string s, Trie* node, int length){
        //剪枝
        if(t->cnt == 0) return;
        
        char ch = board[x][y];
        int r = board.size(), c = board[0].size();

        if(node->children[ch - 'a'] == nullptr) return;

        node = node->children[ch - 'a'];
        
        s+=ch;
        board[x][y] = '.';

        //剪枝
        if(node->isEnd){
            res.push_back(s);
            node->isEnd = false;
            t->cnt--;
        }

        length++;
        if(length > l+1) return;


        for (int i = 0; i < 4; ++i) {
            int nx = x + dirs[i][0];
            int ny = y + dirs[i][1];
            if (nx >= 0 && nx < board.size() && ny >= 0 && ny < board[0].size()) {
                if (board[nx][ny] != '.') {
                    dfs(board, nx, ny, s, node, length);
                }
            }
        }

        board[x][y] = ch;

    }
};
~~~
