## [247. 中心对称数 II](https://leetcode.cn/problems/strobogrammatic-number-ii/)

~~~c++
class Solution {
public:
    vector<char> one_num = {'0', '1', '8'};
    vector<pair<char,char>> pair_num = {{'0','0'},{'1','1'},{'6','9'},{'8','8'},{'9','6'}};

    vector<string> findStrobogrammatic(int n) {
        if(n == 1)
            return {"0","1","8"};

        string mould = "000000000000000";
        mould = mould.substr(0,n);
        vector<string> res;
        //初始化res,将第0位和最后一位赋值
        for(int j = 1; j < 5; j++){
            mould[0] = pair_num[j].first;
            mould[n-1] = pair_num[j].second;
            res.push_back(mould);
        }
        for(int i = 1; i < n / 2; i++){
            //不断将res中所有的字符串取出，然后在第i位和n-i-1位上赋予值
            vector<string> tmp;
            while(!res.empty()){
                string a = res.back();
                res.pop_back();
                //将a的i位和n-i-1位赋值
                for(int j = 0; j < 5; j++){
                    a[i] = pair_num[j].first;
                    a[n-i-1] = pair_num[j].second;
                    tmp.push_back(a);
                }
            }
            swap(tmp, res);
        }
        //最后如果n为奇数还需将中间的数字做变化
        if(n % 2){
            vector<string> tmp;
            while(!res.empty()){
                string a = res.back();
                res.pop_back();
                //将中心赋值
                for(int j = 0; j < 3; j++){
                    a[n / 2] = one_num[j];
                    tmp.push_back(a);
                }
            }
            return tmp;
        }

        return res;
    }
};
~~~
