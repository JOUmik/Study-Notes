## 17. 电话号码的字母组合

**关键词**：回溯， DFS

~~~c++
class Solution {
public:
    vector<string> digitalTable;
    vector<string> res;

    vector<string> letterCombinations(string digits) {
        digitalTable.push_back("abc");
        digitalTable.push_back("def");
        digitalTable.push_back("ghi");
        digitalTable.push_back("jkl");
        digitalTable.push_back("mno");
        digitalTable.push_back("pqrs");
        digitalTable.push_back("tuv");
        digitalTable.push_back("wxyz");

        if(digits.size() == 0){
            return res;
        }

        string s;

        dfs(digits, 0, s);

        return res;
    }

    void dfs(string digits, int index, string s){
        string nset = digitalTable[digits[index] - '2'];
        for(int i = 0; i < nset.size(); i++){
            s += nset[i];

            if(digits.size() - 1 != index){
                dfs(digits, index+1, s);
            }
            else{
                res.push_back(s);
            }

            s = s.substr(0, s.size() - 1);
        }
    }
};
~~~



## 77. 组合

**关键词**：回溯，DFS

~~~c++
class Solution {
public:
    vector<vector<int>> res;

    vector<vector<int>> combine(int n, int k) {
        vector<int> v;
        dfs(v, n, k);

        return res;
    }

    void dfs(vector<int>& v, int n, int k){
        int begin = 1;
        if(!v.empty()) begin = v[v.size()-1] + 1;
        
        for(int i = begin; i <= n; i++){
            v.push_back(i);

            if(v.size() != k){
                dfs(v, n, k);
            }
            else{
                res.push_back(v);
            }

            v.pop_back();
        }
    }
};
~~~
