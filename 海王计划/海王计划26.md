## [624. 数组列表中的最大距离](https://leetcode.cn/problems/maximum-distance-in-arrays/)

~~~c++
class Solution {
public:
    int maxDistance(vector<vector<int>>& arrays) {
        int low = arrays[0][0];
        int high = arrays[0][arrays[0].size()-1];
        int res = 0;

        for(int i = 1; i < arrays.size(); i++){
            int n = arrays[i].size();
            int i_l = arrays[i][0], i_r = arrays[i][n-1];

            res = max(res, max(high - i_l, i_r - low));

            high = max(high, i_r);
            low = min(low, i_l);
        }

        return res;
    }
};
~~~



## [280. 摆动排序](https://leetcode.cn/problems/wiggle-sort/)

~~~c++
class Solution {
public:
    void wiggleSort(vector<int>& nums) {
        bool down = true;
        for(int i = 0; i < nums.size() - 1; i++){
            //底
            if(down){
                if(nums[i] > nums[i+1]){
                    swap(nums[i], nums[i+1]);   
                }
            }
            //顶
            else{
                if(nums[i] < nums[i+1]){
                    swap(nums[i], nums[i+1]);
                }
            }
            down = !down;
        }
    }
};
~~~



## [1056. 易混淆数](https://leetcode.cn/problems/confusing-number/)

~~~c++
class Solution {
public:
    bool confusingNumber(int n) {
        int origin = n, reverse = 0;
        while(n != 0){
            int remainder = n%10;
            n/=10;
            if(remainder == 0) reverse = reverse * 10 + 0;
            else if(remainder == 1) reverse = reverse * 10 + 1;
            else if(remainder == 6) reverse = reverse * 10 + 9;
            else if(remainder == 8) reverse = reverse * 10 + 8;
            else if(remainder == 9) reverse = reverse * 10 + 6;
            else return false;
        }

        if(reverse != origin) return true;
        return false;
    }
};
~~~
