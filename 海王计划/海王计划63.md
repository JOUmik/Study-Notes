# 海王计划63



## [1231. 分享巧克力](https://leetcode.cn/problems/divide-chocolate/)

**关键词：**二分查找

~~~c++
class Solution {
public:
   // 是否可以把巧克力切成甜度不低于mid的k+1块
   bool check(vector<int>& sweetness, int K ,int mid) {
       int cnt = 0;
       int sum = 0;
       for(int s:sweetness){
           sum += s;
           if(sum >= mid){
               cnt++;
               sum = 0;
           }
       }
       return cnt >= K+1;
   }

   int maximizeSweetness(vector<int>& sweetness, int K) {
       int left = *min_element(sweetness.begin(),sweetness.end());
       int right = accumulate(sweetness.begin(),sweetness.end(), 0)/(K+1);
       int ans = 0;
       while(left <= right){
           int mid = (left + right) >>1;
           if(check(sweetness,K,mid)){
               ans = mid; // 更新 ans
               left = mid + 1; // 将 left 移动到 mid 的右边一个位置，寻找更大的甜度值
           }else {
               right = mid-1;
           }
       }
       return ans;
   }
};
~~~
