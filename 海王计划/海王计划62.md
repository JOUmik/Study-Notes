## [265. 粉刷房子 II](https://leetcode.cn/problems/paint-house-ii/)

**关键词：** 动态规划

~~~
class Solution {
public:
    int minCostII(vector<vector<int>>& costs) {
        int n = costs.size();
        int m = costs[0].size();

        int MinIndex = 0;

        
        for(int i = 1; i<m; i++){
            if(costs[0][MinIndex] > costs[0][i]){
                MinIndex = i;
            }
        } 

        for(int i = 1; i < n; i++){
            int MinIndex_i = 0;
            for(int j = 0; j < m; j++){
                if(MinIndex != j){
                    costs[i][j] += costs[i-1][MinIndex];
                }
                else{
                    int CostMin = INT_MAX;
                    for(int z = 0; z < m; z++){
                        if(z == j) continue;
                        CostMin = min(CostMin, costs[i-1][z]);
                    }
                    costs[i][j] += CostMin;
                }

                if(costs[i][MinIndex_i] > costs[i][j]){
                    MinIndex_i = j;
                }
            }
            MinIndex = MinIndex_i;
        }

        return costs[n-1][MinIndex];
    }
};
~~~



## [1259. 不相交的握手](https://leetcode.cn/problems/handshakes-that-dont-cross/)

**关键词**： 动态规划

~~~c++
class Solution {
public:
    int numberOfWays(int numPeople) {
        long n = numPeople / 2;
        vector<long long> cnt(n, 0);
        cnt[0] = 1;
        for(int i = 1; i<n; i++){
            long long cnt_i = 0;
            for(int j = i; j>=0; j--){
                long long pickcnt = 1;
                int index_l = j - 0 - 1;
                int index_r = i - j - 1;
                if(index_l >= 0) pickcnt *= cnt[index_l];
                if(index_r >= 0) pickcnt *= cnt[index_r];

                cnt_i+=pickcnt;

                cnt_i %= 1000000007;
            }

            cnt[i] = cnt_i;
        }

        return cnt[n-1] > (1000000007) ? cnt[n-1] % (1000000007) : cnt[n-1];
    }
};
~~~
