## [373. 查找和最小的 K 对数字](https://leetcode.cn/problems/find-k-pairs-with-smallest-sums/)

**关键词**：Lambda表达式， 优先队列

~~~c++
class Solution {
public:
    vector<vector<int>> kSmallestPairs(vector<int>& nums1, vector<int>& nums2, int k) {
        //Lambda表达式
        auto cmp = [&nums1, &nums2](const pair<int, int>& a, const pair<int, int> & b){
            return nums1[a.first] + nums2[a.second] > nums1[b.first] + nums2[b.second];
        };

        int m = nums1.size(), n = nums2.size();

        vector<vector<int>> ans;
        priority_queue<pair<int, int>, vector<pair<int, int>>, decltype(cmp)> pq(cmp);
        for(int i = 0; i < min(m, k); i++){
            pq.emplace(i, 0);
        }

        while(k-- > 0 && !pq.empty()){
            auto [x, y] = pq.top();
            pq.pop();
            ans.emplace_back(vector<int>{nums1[x], nums2[y]});

            if(y + 1 < n){
                pq.emplace(x, y + 1);
            }
        }

        return ans;

    }
};
~~~



##  [295. 数据流的中位数](https://leetcode.cn/problems/find-median-from-data-stream/)

**关键词**：优先队列

~~~c++
class MedianFinder {
public:
    MedianFinder() {

    }
    
    void addNum(int num) {
        if(Minq.empty() || num <= Minq.top()){
            Minq.push(num);
            if(Maxq.size() + 1 < Minq.size()){
                Maxq.push(Minq.top());
                Minq.pop();
            }
        }
        else{
            Maxq.push(num);
            if(Maxq.size() > Minq.size()){
                Minq.push(Maxq.top());
                Maxq.pop();
            }
        }
    }
    
    double findMedian() {
        if(Minq.size() > Maxq.size()){
            return Minq.top();
        }
        return (Maxq.top() + Minq.top()) / 2.0;
    }

    priority_queue<int, vector<int>, less<int>> Minq;
    priority_queue<int, vector<int>, greater<int>> Maxq;
};

/**
 * Your MedianFinder object will be instantiated and called as such:
 * MedianFinder* obj = new MedianFinder();
 * obj->addNum(num);
 * double param_2 = obj->findMedian();
 */
~~~
