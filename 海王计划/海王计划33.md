## [616. 给字符串添加加粗标签](https://leetcode.cn/problems/add-bold-tag-in-string/)

~~~c++
class Solution {
public:
    string addBoldTag(string s, vector<string>& words) {
        vector<bool> cnt(s.size());

        for(string word:words){
            int pos = 0;
            pos = s.find(word, pos);
            while(pos != -1){
                for(int i = pos; i < pos+word.size(); i++){
                    cnt[i] = true;
                }
                pos++;
                pos = s.find(word, pos);
            }
        }

        string res="";
        for(int i = 0; i<s.size(); i++){
            if (cnt[i] && (i == 0 || !cnt[i-1])) {
                res.append("<b>");
            }
            res.push_back(s[i]);
            if (cnt[i] && (i == s.size()-1 || !cnt[i+1])) {
                res.append("</b>");
            }
        }

        return res;
    }
};
~~~



##  [1272. 删除区间](https://leetcode.cn/problems/remove-interval/)

~~~c++
class Solution {
public:
    vector<vector<int>> removeInterval(vector<vector<int>>& intervals, vector<int>& toBeRemoved) {
        vector<vector<int>> res;

        for(int i = 0; i<intervals.size(); i++){
            if(intervals[i][1] <= toBeRemoved[0] || intervals[i][0] >= toBeRemoved[1]){
                res.push_back(intervals[i]);
                continue;
            }
            if(intervals[i][0] < toBeRemoved[0] && intervals[i][1] > toBeRemoved[1]){
                res.push_back(vector<int>{intervals[i][0], toBeRemoved[0]});
                res.push_back(vector<int>{toBeRemoved[1], intervals[i][1]});
            }
            else if(intervals[i][0] < toBeRemoved[0] && intervals[i][1] > toBeRemoved[0]){
                res.push_back(vector<int>{intervals[i][0], toBeRemoved[0]});
            }
            else if(intervals[i][0] < toBeRemoved[1] && intervals[i][1] > toBeRemoved[1]){
                res.push_back(vector<int>{toBeRemoved[1], intervals[i][1]});
            }
        }

        return res;
    }
};
~~~
