## [1060. 有序数组中的缺失元素](https://leetcode.cn/problems/missing-element-in-sorted-array/)

~~~c++
class Solution {
public:
    int missing(const vector<int>& nums, int index) {
        return nums[index] - nums[0] - index;
    }
    int missingElement(vector<int>& nums, int k) {
        int n = nums.size();
        if(k > missing(nums,n - 1)) {
            return nums[n - 1] + k - missing(nums, n - 1);
        }
        int pos = 1;
        while(missing(nums, pos) < k) pos++;
        return nums[pos - 1] + k - missing(nums, pos - 1);
    }
};
~~~
