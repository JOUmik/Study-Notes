## [1086. 前五科的均分](https://leetcode.cn/problems/high-five/)

~~~c++
class Solution {
public:
    vector<vector<int>> highFive(vector<vector<int>>& items) {
        map<int, priority_queue<int, vector<int>, less<int>>> umap;

        for(int i = 0; i<items.size(); i++){
            umap[items[i][0]].emplace(items[i][1]);
        }
        vector<vector<int>> res;

        for(auto& it : umap){
            int average = 0;
            for(int i = 0; i < 5; i++){
                average += it.second.top();
                it.second.pop();
            }

            average /= 5;
            res.push_back({it.first, average});
        }

        return res;
    }
};
~~~



## [1167. 连接木棍的最低费用](https://leetcode.cn/problems/minimum-cost-to-connect-sticks/)

~~~c++
class Solution {
public:
    int connectSticks(vector<int>& sticks) {
        int totalCost = 0;
       
        priority_queue<int, vector<int>, greater<int>> pq;
        
        // 所有棍子都添加到最小堆中
        for (int i = 0; i < sticks.size(); i++) {
            pq.push(sticks[i]);
        }
        
        // 把两根最小的棍子合起来，直到只剩下一根。
        while (pq.size() > 1) {
            int stick1 = pq.top(); 
            pq.pop();
            int stick2 = pq.top(); 
            pq.pop();
            
            int cost = stick1 + stick2;
            totalCost += cost;
            
            pq.push(stick1+stick2);
        }
        
        return totalCost;
    }
};
~~~
