## [94. 二叉树的中序遍历](https://leetcode.cn/problems/binary-tree-inorder-traversal/)

~~~c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> res;
    vector<int> inorderTraversal(TreeNode* root) {
        if(!root) return res;
        if(root->left){
            inorderTraversal(root->left);
        }
        res.push_back(root->val);
        if(root->right){
            inorderTraversal(root->right);
        }

        return res;
    }
};
~~~



## [437. 路径总和 III](https://leetcode.cn/problems/path-sum-iii/)

~~~c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int res = 0;
    int pathSum(TreeNode* root, int targetSum) {
        if(!root) return res;
        vector<long> cnt;
    
        calculate(root, targetSum, cnt);
        return res;

    }

    int calculate(TreeNode* root, int targetSum, vector<long> cnt){
        if(!root) return res;
        for(int i = 0; i<cnt.size(); i++){
            cnt[i]+= root->val;
            if(cnt[i] == targetSum){
                res++;
            }
        }
        cnt.push_back(root->val);
        if(cnt[cnt.size()-1] == targetSum) res++;
        calculate(root->left, targetSum, cnt);
        calculate(root->right, targetSum, cnt);
        return res;
    }

};
~~~



## [994. 腐烂的橘子](https://leetcode.cn/problems/rotting-oranges/)

**关键词**：模拟

~~~c++
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int res = 0;
        int freshNum = 0;
        for(int i = 0; i<grid.size(); i++){
            for(int j = 0; j<grid[0].size(); j++){
                if(grid[i][j] == 1) freshNum++;
            }
        }
        int LastFresh = freshNum;
        int coruptindex = 3;
        while(freshNum!=0)
        {
            for(int i = 0; i<grid.size(); i++){
                for(int j = 0; j<grid[0].size(); j++){
                    if(grid[i][j] == coruptindex-1){
                        if(i>0 && grid[i-1][j] == 1){
                            grid[i-1][j] = coruptindex;
                            freshNum--;
                        } 
                        if(j>0 && grid[i][j-1] == 1){
                            grid[i][j-1] = coruptindex;
                            freshNum--;
                        } 
                        if(i<grid.size()-1 && grid[i+1][j] == 1){
                            grid[i+1][j] = coruptindex;
                            freshNum--;
                        } 
                        if(j<grid[0].size()-1 && grid[i][j+1] == 1){
                            grid[i][j+1] = coruptindex;
                            freshNum--;
                        } 
                    }
                }
            }
            //如果循环一圈没有橘子腐败说明永远不会腐烂
            if(freshNum == LastFresh) return -1;

            LastFresh = freshNum;
            coruptindex++;
            res++;
        }

        return res;
    }
};
~~~
