## [723. 粉碎糖果](https://leetcode.cn/problems/candy-crush/)

~~~c++
class Solution {
public:
    vector<vector<int>> candyCrush(vector<vector<int>>& board) {
        int r = board.size(), c = board[0].size();
        bool cnt = false;

        for(int i = 0; i < r; i++){
            for(int j = 0; j + 2 < c; j++){
                int val = abs(board[i][j]);
                if(val && val == abs(board[i][j+1]) && val == abs(board[i][j+2])){
                    board[i][j] = -val;
                    board[i][j+1] = -val;
                    board[i][j+2] = -val;
                    cnt = true;
                }
            }
        }

        for(int i = 0; i + 2 < r; i++){
            for(int j = 0; j < c; j++){
                int val = abs(board[i][j]);
                if(val && val == abs(board[i+1][j]) && val == abs(board[i+2][j])){
                    board[i][j] = -val;
                    board[i+1][j] = -val;
                    board[i+2][j] = -val;
                    cnt = true;
                }
            }
        }

        for(int j = 0; j < c; j++){
            int wr = r - 1;
            for(int i = r - 1; i >= 0; i--){
                if(board[i][j] > 0){
                    board[wr][j] = board[i][j];
                    wr--;
                }
            }

            while(wr>=0){
                board[wr][j] = 0;
                wr--;
            }
        }
        

        return cnt ? candyCrush(board) : board; 
    }
};
~~~



## [163. 缺失的区间](https://leetcode.cn/problems/missing-ranges/)

~~~c++
class Solution {
public:
    vector<vector<int>> findMissingRanges(vector<int>& nums, int lower, int upper) {
        vector<vector<int>> res;
        if(nums.size() == 0){
            res.emplace_back(vector<int>{lower, upper});
            return res;
        }

         if(nums[0] > lower) res.emplace_back(vector<int>{lower, nums[0]-1});
        for(int i = 0; i + 1<nums.size(); i++){
            if(nums[i+1] - nums[i] >= 2){
                res.emplace_back(vector<int>{nums[i]+1, nums[i+1]-1});
            }
        }
        if(nums[nums.size()-1] < upper) res.emplace_back(vector<int>{nums[nums.size()-1] + 1, upper});

        return res;
    }
};
~~~
